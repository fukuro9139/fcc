57 {
58  /* 左辺がポインタなら同じ型へのポインタを返す。 */
59  if (ty1->_base)
60  {
61  return pointer_to(ty1->_base);
62  }
63 
64  /* 演算の中に出てくる関数は関数ポインタ */
65  if (TypeKind::TY_FUNC == ty1->_kind)
66  {
67  return pointer_to(ty1);
68  }
69  if (TypeKind::TY_FUNC == ty2->_kind)
70  {
71  return pointer_to(ty2);
72  }
73 
74  /* 片方がdouble型ならdouble型 */
75  if (TypeKind::TY_DOUBLE == ty1->_kind || TypeKind::TY_DOUBLE == ty2->_kind)
76  {
77  return Type::DOUBLE_BASE;
78  }
79 
80  /* 片方がfloat型ならfloat型 */
81  if (TypeKind::TY_FLOAT == ty1->_kind || TypeKind::TY_FLOAT == ty2->_kind)
82  {
83  return Type::FLOAT_BASE;
84  }
85 
86  /* 算術演算の結果はint型以上とする */
87  if (ty1->_size < 4)
88  {
89  ty1 = Type::INT_BASE;
90  }
91  if (ty2->_size < 4)
92  {
93  ty2 = Type::INT_BASE;
94  }
95 
96  /* サイズが違えばサイズが大きい方を返す */
97  if (ty1->_size != ty2->_size)
98  {
99  return (ty1->_size < ty2->_size) ? ty2 : ty1;
100  }
101 
102  /* サイズが同じ場合は右オペランドがunsignedならunsignedを返す */
103  if (ty2->_is_unsigned)
104  {
105  return ty2;
106  }
107 
108  return ty1;
109 }

116 void Type::add_type(Node *node)
117 {
118  /* ノードが存在しない or 既に型が設定されている場合は何もしない */
119  if (!node || node->_ty)
120  {
121  return;
122  }
123 
124  add_type(node->_lhs.get());
125  add_type(node->_rhs.get());
126  add_type(node->_condition.get());
127  add_type(node->_then.get());
128  add_type(node->_else.get());
129  add_type(node->_init.get());
130  add_type(node->_inc.get());
131 
132  for (auto n = node->_body.get(); n; n = n->_next.get())
133  {
134  add_type(n);
135  }
136  for (auto n = node->_args.get(); n; n = n->_next.get())
137  {
138  add_type(n);
139  }
140 
141  switch (node->_kind)
142  {
143  /* 演算結果の型は大きいほうの型にあわせる */
144  case NodeKind::ND_ADD:
145  case NodeKind::ND_SUB:
146  case NodeKind::ND_MUL:
147  case NodeKind::ND_DIV:
148  usual_arith_conv(node->_lhs, node->_rhs);
149  node->_ty = node->_lhs->_ty;
150  break;
151 
152  case NodeKind::ND_NEG:
153  {
154  auto ty = get_common_type(INT_BASE, node->_lhs->_ty);
155  node->_lhs = Node::new_cast(move(node->_lhs), ty);
156  node->_ty = ty;
157  break;
158  }
159 
160  case NodeKind::ND_ASSIGN:
161  if (TypeKind::TY_ARRAY == node->_lhs->_ty->_kind)
162  {
163  error_token("左辺値ではありません", node->_lhs->_token);
164  }
165  /* 左辺に合わせてキャストする */
166  if (TypeKind::TY_STRUCT != node->_lhs->_ty->_kind)
167  {
168  node->_rhs = Node::new_cast(move(node->_rhs), node->_lhs->_ty);
169  }
170  node->_ty = node->_lhs->_ty;
171  break;
172 
173  case NodeKind::ND_COND:
174  /* どちらかがvoid型の場合はvoid型 */
175  if (TypeKind::TY_VOID == node->_then->_ty->_kind ||
176  TypeKind::TY_VOID == node->_else->_ty->_kind)
177  {
178  node->_ty = Type::VOID_BASE;
179  }
180  /* 大きい方の型に合わせる */
181  else
182  {
183  usual_arith_conv(node->_then, node->_else);
184  node->_ty = node->_then->_ty;
185  }
186  break;
187 
188  /* カンマ区切りの型は右辺に一致させる */
189  case NodeKind::ND_COMMA:
190  node->_ty = node->_rhs->_ty;
191  break;
192 
193  /* 構造体の型はメンバの型 */
194  case NodeKind::ND_MEMBER:
195  node->_ty = node->_member->_ty;
196  break;
197 
198  /* 参照は参照先へのポインタ型 */
199  case NodeKind::ND_ADDR:
200  {
201  auto ty = node->_lhs->_ty;
202 
203  /* 配列型の変数への参照はポインタ型としてみた配列の型と同じ */
204  if (TypeKind::TY_ARRAY == ty->_kind)
205  {
206  node->_ty = pointer_to(node->_lhs->_ty->_base);
207  }
208  /* そうでなければ左辺の型へのポインタ */
209  else
210  {
211  node->_ty = pointer_to(ty);
212  }
213  break;
214  }
215 
216  /* デリファレンス */
217  case NodeKind::ND_DEREF:
218  /* デリファレンスできるのはポインタ型のみ */
219  if (!node->_lhs->_ty->_base)
220  {
221  error_token("デリファレンスできるのはポインタ型のみです", node->_token);
222  }
223  /* void型のポインタはデリファレンス不可 */
224  if (TypeKind::TY_VOID == node->_lhs->_ty->_base->_kind)
225  {
226  error_token("void型のポインタはデリファレンスできません", node->_token);
227  }
228  /* ポインタのベース型 */
229  node->_ty = node->_lhs->_ty->_base;
230  break;
231 
232  case NodeKind::ND_NOT:
233  node->_ty = INT_BASE;
234  break;
235 
236  case NodeKind::ND_BITNOT:
237  node->_ty = node->_lhs->_ty;
238  break;
239 
240  case NodeKind::ND_LOGAND:
241  case NodeKind::ND_LOGOR:
242  node->_ty = INT_BASE;
243  break;
244 
245  case NodeKind::ND_FUNCALL:
246  node->_ty = Type::LONG_BASE;
247  break;
248 
249  case NodeKind::ND_MOD:
250  case NodeKind::ND_BITAND:
251  case NodeKind::ND_BITOR:
252  case NodeKind::ND_BITXOR:
253  usual_arith_conv(node->_lhs, node->_rhs);
254  node->_ty = node->_lhs->_ty;
255  break;
256 
257  case NodeKind::ND_SHL:
258  case NodeKind::ND_SHR:
259  node->_ty = node->_lhs->_ty;
260  break;
261 
262  /* 比較演算は大きい方の型に合わせて行う、結果はint型とする */
263  case NodeKind::ND_EQ:
264  case NodeKind::ND_NE:
265  case NodeKind::ND_LT:
266  case NodeKind::ND_LE:
267  /* 比較の前に大きいほうの型に合わせる */
268  usual_arith_conv(node->_lhs, node->_rhs);
269  node->_ty = Type::INT_BASE;
270  break;
271 
272  case NodeKind::ND_NUM:
273  node->_ty = INT_BASE;
274  break;
275 
276  /* ステートメント式 */
277  case NodeKind::ND_STMT_EXPR:
278  {
279  /* 中身がなければ何もしない */
280  if (!node->_body)
281  {
282  break;
283  }
284  auto stmt = node->_body.get();
285  /* 複文の中を最後まで辿っていく */
286  while (stmt->_next)
287  {
288  stmt = stmt->_next.get();
289  }
290 
291  if (NodeKind::ND_EXPR_STMT == stmt->_kind)
292  {
293  node->_ty = stmt->_lhs->_ty;
294  break;
295  }
296  error_token("ステートメント式はvoid型の戻り値をサポートしていません", node->_token);
297  }
298 
299  case NodeKind::ND_VAR:
300  /* オブジェクトの型に一致させる */
301  node->_ty = node->_var->_ty;
302  break;
303 
304  default:
305  break;
306  }
