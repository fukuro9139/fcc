108 void CodeGen::load(const Type *ty)
109 {
110  /* 変数の型が配列型の場合、レジスタへ値をロードをしようとしてはいけない。*/
111  /* これは一般的に配列全体をレジスタに読み込むことはできないからである。*/
112  /* そのため配列型の変数の評価は配列そのものではなく配列のアドレスとなる。*/
113  /* これは配列型が暗黙に配列の最初の要素へのポインタへ返還されることを示している。 */
114  switch (ty->_kind)
115  {
116  case TypeKind::TY_FLOAT:
117  *os << " movss xmm0, DWORD PTR [rax]\n";
118  return;
119  case TypeKind::TY_DOUBLE:
120  *os << " movsd xmm0, QWORD PTR [rax]\n";
121  return;
122  case TypeKind::TY_FUNC:
123  case TypeKind::TY_ARRAY:
124  case TypeKind::TY_STRUCT:
125  case TypeKind::TY_UNION:
126  return;
127  default:
128  break;
129  }
130 
131  /* char型およびshort型の値をロードするとき、常にint型のサイズに拡張する。
132  * つまり、64ビットレジスタの下位32ビットは常に正しい値が入っているとみなせる。
133  * このとき上位32ビットレジスタにはごみが残っている可能性がある。
134  * int型よりサイズの大きな型の値をロードするときは単純にレジスタ全体が置き換えられる。
135  */

141  }

143  {
144  *os << insn << " eax, WORD PTR [rax]\n";
145  }

475 void CodeGen::generate_expression(Node *node)
476 {
477  if (print_dbg_info)
478  {
479  *os << " .loc " << node->_token->_file->_file_no << " " << node->_token->_line_no << "\n";
480  }
481 
482  switch (node->_kind)
483  {
484  /* NULL */
485  case NodeKind::ND_NULL_EXPR:
486  break;
487 
488  /* 単項演算子の'-' */
489  case NodeKind::ND_NEG:
490  /* '-'がかかる式を評価 */
491  generate_expression(node->_lhs.get());
492 
493  switch (node->_ty->_kind)
494  {
495  case TypeKind::TY_FLOAT:
496  /* 符号ビット(32ビット)を反転 */
497  *os << " mov rax, 1\n";
498  *os << " shl rax, 31\n";
499  *os << " movq xmm1, rax\n";
500  *os << " xorps xmm0, xmm1\n";
501  break;
502  case TypeKind::TY_DOUBLE:
503  *os << " mov rax, 1\n";
504  *os << " shl rax, 63\n";
505  *os << " movq xmm1, rax\n";
506  *os << " xorpd xmm0, xmm1\n";
507  break;
508  default:
509  /* 符号を反転させる */
510  *os << " neg rax\n";
511  }
512  break;
513 
514  /* 代入 */
515  case NodeKind::ND_ASSIGN:
516  /* 左辺の代入先の変数のアドレスを計算 */
517  generate_address(node->_lhs.get());
518  /* 変数のアドレスをスタックにpush */
519  push();
520  /* 右辺を評価する。評価結果は'rax' */
521  generate_expression(node->_rhs.get());
522  /* raxの値をストアする */
523  store(node->_ty.get());
524  break;
525 
526  /* 3項演算子 */
527  case NodeKind::ND_COND:
528  {
529  const int c = label_count();
530  /* 条件 */
531  generate_expression(node->_condition.get());
532  cmp_zero(node->_condition->_ty.get());
533  *os << " je .L.else." << c << "\n";
534  /* 条件が真のとき */
535  generate_expression(node->_then.get());
536  *os << " jmp .L.end." << c << "\n";
537  /* 条件が偽の時 */
538  *os << ".L.else." << c << ":\n";
539  generate_expression(node->_else.get());
540  *os << ".L.end." << c << ":\n";
541  break;
542  }
543 
544  /* カンマ区切り */
545  case NodeKind::ND_COMMA:
546  generate_expression(node->_lhs.get());
547  generate_expression(node->_rhs.get());
548  break;
549 
550  /* 構造体のメンバ */
551  case NodeKind::ND_MEMBER:
552  {
553  /* 変数のアドレスを計算 */
554  generate_address(node);
555  /* 変数のアドレスから値をロードする */
556  load(node->_ty.get());
557  break;
558  }
559 
560  /* 参照 */
561  case NodeKind::ND_ADDR:
562  /* アドレスを計算 */
563  generate_address(node->_lhs.get());
564  break;
565 
566  /* 参照外し */
567  case NodeKind::ND_DEREF:
568  /* 参照先のアドレスを計算 */
569  generate_expression(node->_lhs.get());
570  /* 参照先のアドレスの値をロード */
571  load(node->_ty.get());
572  break;
573 
574  /* ! */
575  case NodeKind::ND_NOT:
576  generate_expression(node->_lhs.get());
577  /* 0と比較 */
578  cmp_zero(node->_lhs->_ty.get());
579  /* 0と一致なら1, それ以外は0 */
580  *os << " sete al\n";
581  *os << " movzx rax, al\n";
582  break;
583 
584  /* ~ */
585  case NodeKind::ND_BITNOT:
586  generate_expression(node->_lhs.get());
587  *os << " not rax\n";
588  break;
589 
590  /* && */
591  case NodeKind::ND_LOGAND:
592  {
593  const int c = label_count();
594  generate_expression(node->_lhs.get());
595  cmp_zero(node->_lhs->_ty.get());
596  /* 短絡評価、前半がfalseなら後半は評価しない */
597  *os << " je .L.false." << c << "\n";
598  generate_expression(node->_rhs.get());
599  cmp_zero(node->_rhs->_ty.get());
600  *os << " je .L.false." << c << "\n";
601  *os << " mov rax, 1\n";
602  *os << " jmp .L.end." << c << "\n";
603  *os << ".L.false." << c << ":\n";
604  *os << " mov rax, 0\n";
605  *os << ".L.end." << c << ":\n";
606  break;
607  }
608 
609  /* || */
610  case NodeKind::ND_LOGOR:
611  {
612  const int c = label_count();
613  generate_expression(node->_lhs.get());
614  cmp_zero(node->_lhs->_ty.get());
615  /* 短絡評価、前半がtrueなら後半は評価しない */
616  *os << " jne .L.true." << c << "\n";
617  generate_expression(node->_rhs.get());
618  cmp_zero(node->_rhs->_ty.get());
619  *os << " jne .L.true." << c << "\n";
620  *os << " mov rax, 0\n";
621  *os << " jmp .L.end." << c << "\n";
622  *os << ".L.true." << c << ":\n";
623  *os << " mov rax, 1\n";
624  *os << ".L.end." << c << ":\n";
625  break;
626  }
627 
628  /* 関数呼び出し */
629  case NodeKind::ND_FUNCALL:
630  {
631  /* スタックに入れる */
632  int stack_args = push_args(node->_args.get());
633  /* raxに関数のアドレスを入れる */
634  generate_expression(node->_lhs.get());
635 
636  /* 引数の数(gp:整数, fp:浮動小数点数) */
637  int gp = 0, fp = 0;
638 
639  /* 引数をレジスタにセットしていく */
640  for (auto arg = node->_args.get(); arg; arg = arg->_next.get())
641  {
642  if (arg->_ty->is_flonum())
643  {
644  if (fp < FP_MAX)
645  {
646  popf(fp++);
647  }
648  }
649  else
650  {
651  if (gp < GP_MAX)
652  {
653  pop(arg_regs64[gp++]);
654  }
655  }
656  }
657 
658  /* 関数のアドレスをr10に格納 */
659  *os << " mov r10, rax\n";
660  /* fpの数をraxに格納 */
661  *os << " mov rax, " << fp << "\n";
662  /* 関数を呼び出す */
663  *os << " call r10\n";
664 
665  /* スタック経由で渡した引数を片付ける */
666  *os << " add rsp, " << stack_args * 8 << "\n";
667  depth -= stack_args;
668 
669  /* x86-64において関数の戻り値がchar/bool, shortのときraxの上位48 または56ビット
670  * は不定となる。そのため関数の戻り値がchar/bool, shortのとき上位ビットをクリアする。
671  */
672  switch (node->_ty->_kind)
673  {
674  case TypeKind::TY_BOOL:
675  *os << " movzx eax, al\n";
676  break;
677  case TypeKind::TY_CHAR:
678  *os << (node->_ty->_is_unsigned ? " movzx" : " movsx") << " eax, al\n";
679  break;
680  case TypeKind::TY_SHORT:
681  *os << (node->_ty->_is_unsigned ? " movzx" : " movsx") << " eax, ax\n";
682  break;
683  default:
684  break;
685  }
686  break;
687  }
688 
689  /* 数値 */
690  case NodeKind::ND_NUM:
691  {
692  /* 浮動小数点を扱うための共用体 */
693  union
694  {
695  float f32;
696  double f64;
697  uint32_t u32;
698  uint64_t u64;
699  } u;
700 
701  if (TypeKind::TY_FLOAT == node->_ty->_kind)
702  {
703  u.f32 = node->_fval;
704  *os << " mov eax, " << u.u32 << " #float " << node->_fval << "\n";
705  *os << " movq xmm0, rax\n";
706  break;
707  }
708  else if (TypeKind::TY_DOUBLE == node->_ty->_kind)
709  {
710  u.f64 = node->_fval;
711  *os << " mov rax, " << u.u64 << " #double " << node->_fval << "\n";
712  *os << " movq xmm0, rax\n";
713  break;
714  }
715 
716  /* 数値を'rax'に格納 */
717  *os << " mov rax, " << node->_val << "\n";
718  break;
719  }
720 
721  /* 重複文 */
722  case NodeKind::ND_STMT_EXPR:
723  {
724  for (auto cur = node->_body.get(); cur; cur = cur->_next.get())
725  {
726  generate_statement(cur);
727  }
728  break;
729  }
730 
731  /* 構造体のメンバ */
732  case NodeKind::ND_VAR:
733  {
734  /* メンバのアドレスを計算 */
735  generate_address(node);
736  /* メンバのアドレスから値をロードする */
737  load(node->_ty.get());
738  break;
739  }
740 
741  case NodeKind::ND_CAST:
742  generate_expression(node->_lhs.get());
743  cast(node->_lhs->_ty.get(), node->_ty.get());
744  break;
745 
746  case NodeKind::ND_MEMZERO:
747  /* 変数が使うメモリサイズ */
748  *os << " mov rcx, " << node->_var->_ty->_size << "\n";
749  /* 変数のアドレス */
750  *os << " lea rdi, [rbp - " << node->_var->_offset << "]\n";
751  /* alに0をセット */
752  *os << " mov al, 0\n";
753  /* 0クリア */
754  *os << " rep stosb\n";
755  break;
756 
757  default:
758  generate_expression2(node);
759  }

767 void CodeGen::generate_expression2(Node *node)
768 {
769  /* 左辺が浮動小数点数の場合 */
770  if (node->_lhs->_ty->is_flonum())
771  {
772  /* 右辺を計算 */
773  generate_expression(node->_rhs.get());
774  /* 計算結果をスタックにpush */
775  pushf();
776  /* 左辺を計算 */
777  generate_expression(node->_lhs.get());
778  /* 右辺の計算結果を'rdi'にpop */
779  popf(1);
780 
781  string sz = (TypeKind::TY_FLOAT == node->_lhs->_ty->_kind) ? "ss" : "sd";
782 
783  switch (node->_kind)
784  {
785  case NodeKind::ND_ADD:
786  /* 'xmm0' = 'xmm0' + 'xmm1' */
787  *os << " add" << sz << " xmm0, xmm1\n";
788  break;
789 
790  case NodeKind::ND_SUB:
791  /* 'xmm0' = 'xmm0' - 'xmm1' */
792  *os << " sub" << sz << " xmm0, xmm1\n";
793  break;
794 
795  case NodeKind::ND_MUL:
796  /* 'xmm0' = 'xmm0' * 'xmm1' */
797  *os << " mul" << sz << " xmm0, xmm1\n";
798  break;
799 
800  case NodeKind::ND_DIV:
801  /* 'xmm0' = 'xmm0' / 'xmm1' */

973 void CodeGen::generate_statement(Node *node)
974 {
975  if (print_dbg_info)
976  {
977  *os << " .loc " << node->_token->_file->_file_no << " " << node->_token->_line_no << "\n";
978  }
979 
980  switch (node->_kind)
981  {
982  case NodeKind::ND_RETURN:
983  /* return の後に式が存在する場合、戻り値を評価 */
984  if (node->_lhs)
985  {
986  generate_expression(node->_lhs.get());
987  }
988 
989  /* エピローグまでjmpする */
990  *os << " jmp .L.return." << current_func->_name << "\n";
991  break;
992 
993  case NodeKind::ND_IF:
994  {
995  /* 通し番号を取得 */
996  const int c = label_count();
997 
998  /* 条件を評価 */
999  generate_expression(node->_condition.get());
1000  /* 条件を比較 */
1001  cmp_zero(node->_condition->_ty.get());
1002  /* 条件がfalseなら.L.else.cラベルに飛ぶ */
1003  *os << " je .L.else." << c << "\n";
1004  /* trueのときに実行 */
1005  generate_statement(node->_then.get());
1006  /* elseは実行しない */
1007  *os << " jmp .L.end." << c << "\n";
1008 
1009  /* falseのとき実行 */
1010  *os << ".L.else." << c << ":\n";
1011  if (node->_else)
1012  {
1013  generate_statement(node->_else.get());
1014  }
1015  *os << ".L.end." << c << ":\n";
1016  break;
1017  }
1018 
1019  /* for or while */
1020  case NodeKind::ND_FOR:
1021  {
1022  /* 通し番号を取得 */
1023  const int c = label_count();
1024 
1025  /* forの場合、初期化処理 */
1026  if (node->_init)
1027  {
1028  generate_statement(node->_init.get());
1029  }
1030 
1031  *os << ".L.begin." << c << ":\n";
1032  /* 終了条件判定 */
1033  if (node->_condition)
1034  {
1035  generate_expression(node->_condition.get());
1036  cmp_zero(node->_condition->_ty.get());
1037  *os << " je " << node->_brk_label << "\n";
1038  }
1039  /* forの中身 */
1040  generate_statement(node->_then.get());
1041 
1042  /* continue */
1043  *os << node->_cont_label << ":\n";
1044 
1045  /* 加算処理 */
1046  if (node->_inc)
1047  {
1048  generate_expression(node->_inc.get());
1049  }
1050  *os << " jmp .L.begin." << c << "\n";
1051  *os << node->_brk_label << ":\n";
1052  break;
1053  }
1054 
1055  case NodeKind::ND_DO:
1056  {
1057  int c = label_count();
1058  *os << ".L.begin." << c << ":\n";
1059  generate_statement(node->_then.get());
1060  *os << node->_cont_label << ":\n";
1061  generate_expression(node->_condition.get());
1062  cmp_zero(node->_condition->_ty.get());
1063  *os << " jne .L.begin." << c << "\n";
1064  *os << node->_brk_label << ":\n";
1065  break;
1066  }
1067 
1068  case NodeKind::ND_SWITCH:
1069  {
1070  generate_expression(node->_condition.get());
1071 
1072  auto reg = (8 == node->_condition->_ty->_size) ? "rax" : "eax";
1073 
1074  /* 各ケースの条件の値と一致すればラベルにjump */
1075  for (auto n = node->case_next; n; n = n->case_next)
1076  {
1077  *os << " cmp " << reg << ", " << n->_val << "\n";
1078  *os << " je " << n->_label << "\n";
1079  }
1080  /* defaultがあればdefaultにjump */
1081  if (node->default_case)
1082  {
1083  *os << " jmp " << node->default_case->_label << "\n";
1084  }
1085  /* 一致する数値がなければ抜ける */
1086  *os << " jmp " << node->_brk_label << "\n";
1087 
1088  /* 各ケース */
1089  generate_statement(node->_then.get());
1090  *os << node->_brk_label << ":\n";
1091  break;
1092  }
1093 
1094  case NodeKind::ND_CASE:
1095  *os << node->_label << ":\n";
1096  generate_statement(node->_lhs.get());
1097  break;
1098 
1099  case NodeKind::ND_BLOCK:
1100  {
1101  /* Bodyに含まれるノードをすべて評価する */
1102  for (auto cur = node->_body.get(); cur; cur = cur->_next.get())
1103  {
1104  generate_statement(cur);
1105  }
1106  break;
1107  }
1108 
1109  case NodeKind::ND_GOTO:
1110  *os << " jmp " << node->_unique_label << "\n";
1111  break;
1112 
1113  case NodeKind::ND_LABEL:
1114  *os << node->_unique_label << ":\n";
1115  generate_statement(node->_lhs.get());
1116  break;
1117 
1118  case NodeKind::ND_EXPR_STMT:
1119  generate_expression(node->_lhs.get());
1120  break;
1121 
1122  default:
1123  error_token("不正な構文です", node->_token);
1124  }
