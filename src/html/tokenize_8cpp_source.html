154  auto current_token = head.get();
155  auto itr = current_file->_contents.cbegin();
156  const auto first = current_file->_contents.cbegin();
157  const auto last = current_file->_contents.cend();
158 
159  /* フラグをセット */
160  at_begining = true;
161  has_space = false;
162 
163  while (itr != last)
164  {
165  /* 改行 */
166  if ('\n' == *itr)
167  {
168  at_begining = true;
169  has_space = false;
170 
171  ++itr;
172  continue;
173  }
174 
175  /* 空白文字をスキップ */
176  if (std::isspace(*itr))
177  {
178  ++itr;
179  has_space = true;
180  continue;
181  }
182 
183  /* 行コメント */
184  /* 入力文字列の末尾は'\n'であることが保証されているため */
185  /* itr+1 != lastのチェックは不要 */
186  if ('/' == *itr && '/' == *(itr + 1))
187  {
188  itr += 2;
189  while ('\n' != *itr)
190  {
191  ++itr;
192  }
193  continue;
194  }
195 
196  /* ブロックコメント */
197  if ('/' == *itr && '*' == *(itr + 1))
198  {
199  itr += 2;
200  while (itr != last && !('*' == *itr && '/' == *(itr + 1)))
201  {
202  ++itr;
203  }
204  if (itr == last)
205  {
206  error_at("ブロックコメントが閉じられていません", itr - first);
207  }
208  itr += 2;
209  has_space = true;
210  continue;
211  }
212

215  {

217  current_token = current_token->_next.get();
218  itr += current_token->_str.size();
219  continue;
220  }
221 
222  /* 文字列リテラル */
223  if ('"' == *itr)
224  {
225  current_token->_next = read_string_literal(itr);
226  current_token = current_token->_next.get();
227  continue;
228  }
229 
230  /* 文字リテラル */
231  if ('\'' == *itr)
232  {
233  current_token->_next = read_char_literal(itr, itr);
234  current_token = current_token->_next.get();
235  itr += current_token->_str.size();
236  continue;
237  }
238 
239  /* ワイド文字リテラル */
240  if ('L' == *itr && '\'' == *(itr + 1))
241  {
242  current_token->_next = read_char_literal(itr, itr + 1);
243  current_token = current_token->_next.get();
244  itr += current_token->_str.size();
245  continue;
246  }
247 
248  /* 変数 */
249  if (is_first_char_of_ident(*itr))
250  {
251  /* 先頭は現在のイテレーター */
252  const auto start = itr;
253  /* 1文字ずつ識別子となりうる文字かみていく */
254  /* 識別子となりえない文字が出てくるまで1つの識別子として認識する */
255  do
256  {
257  ++itr;
258  } while (is_char_of_ident(*itr));
259

261  current_token->_next = make_unique<Token>(TokenKind::TK_IDENT, start - first, string(start, itr));
262  current_token = current_token->_next.get();
263  continue;
264  }

268  if (punct_len)

270  /* 新しいトークンを生成してcurに繋ぎcurを一つ進める */
271  current_token->_next = make_unique<Token>(TokenKind::TK_PUNCT, itr - first, string(itr, itr + punct_len));
272  current_token = current_token->_next.get();

274  continue;

278  }
279

282  /* 行数をセットする */
